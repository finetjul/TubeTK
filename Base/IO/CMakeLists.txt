project( TubeIO )

find_package( ITK REQUIRED )
include( ${ITK_USE_FILE} )

set( TubeTK_Base_IO_H_Files
  itkTubeMetaLDA.h
  itkTubeMetaNJetLDA.h
  )

set( TubeTK_Base_IO_TXX_Files
  )

include_directories(
  ${TubeTK_SOURCE_DIR}/Base/Common
  )

set( TubeTK_Base_IO_SRCS
  itkTubeMetaLDA.cxx
  itkTubeMetaNJetLDA.cxx
  )

add_library( TubeIO STATIC
  ${TubeTK_Base_IO_H_Files}
  ${TubeTK_Base_IO_TXX_Files}
  ${TubeTK_Base_IO_SRCS}
  )

target_link_libraries( TubeIO ${ITK_LIBRARIES} )

slicer4_set_plugins_output_path( TubeIO )
set( TARGETS TubeIO )
slicer4_install_plugins( ${TARGETS} )

if( BUILD_TESTING )
  add_subdirectory( Testing )
endif( BUILD_TESTING )

install( FILES
  ${TubeTK_Base_Numerics_H_Files}
  ${TubeTK_Base_Numerics_TXX_Files}
  DESTINATION include
  )

# Export target
set_property( GLOBAL APPEND PROPERTY TubeTK_TARGETS TubeIO )
